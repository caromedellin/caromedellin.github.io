firstFactorial
        function firstFactorial(num){
           if(num === 0){
            num = 1;
           } else {
              for(var i = (num -1); i > 1; i--){
              num *= i;
             }
           }
          return num;
        };

function LetterChanges(str) { var alpha = "abcdefghijklmnopqrstuvwxyz"; var newAlpha = "bcdEfghIjklmnOpqrstUvwxyzA"; var answer = ""; for (i = 0, i < str.length; i++) { if (alpha.indexOf(str[i]) !== -1) { answer += newAlpha.charAt(alpha.indexOf(str[i])); } else { answer += str[i]; } } return answer; }
